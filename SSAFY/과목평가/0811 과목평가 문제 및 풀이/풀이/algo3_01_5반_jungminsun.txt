arr = [5, 1, 4, 2, 8]

n = len(arr) #5

for i in range(n-1, 0, -1): # 4,3,2,1
# 버블 정렬은 뒤에서부터 정렬되고, 정렬을 확인해야하는 범위가 점점 줄어들기 때문에 위 처럼 4, 3, 2, 1로 수정하였다.
    for j in range(i): # 0~3, 0~2, 0~1, 0
        if arr[j] > arr[j+1]:
        # 버블 정렬은 앞에서부터 인접한 값과 비교하는 방식이고, 오름차순으로 정렬하기 위해서는 본인이 뒤의 값보다
        # 클 경우, 뒤로 넘어가면서 정렬되기 때문에 부등호를 변경해주어야 한다.
            arr[j], arr[j+1] = arr[j+1], arr[j]
            # ==의 부등호는 같다라는 의미이므로 = 를 사용하여 값을 변경해주어야 한다.
